local m = require(game:GetService("ReplicatedStorage").LocalEntity)

if not _G.noFallHooked then
    _G.noFallHooked = true

    local old
    old = hookfunction(m, function(state, env, dt)
        if _G.noFall and state then
            state.FallDistance = nil
        end
        return old(state, env, dt)
    end)
end
